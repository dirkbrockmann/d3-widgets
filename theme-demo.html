Code Examples


// Check current dark mode preference
const isDark = widgets.isDarkMode();
console.log(isDark ? "Dark mode" : "Light mode");

// Watch for changes in system color scheme
widgets.watchColorScheme(isDark => {
  console.log(`Theme changed to: ${isDark ? "dark" : "light"}`);
});

// Manually toggle dark mode
const isDarkNow = widgets.toggleDarkMode(); // toggle
widgets.toggleDarkMode(true); // force dark mode
widgets.toggleDarkMode(false); // force light mode
